        // Math to calculate distance between GPS points
        function calculateDistance(lat1, lon1, lat2, lon2) {
            const R = 6371; // Earth radius in km
            const dLat = (lat2 - lat1) * Math.PI / 180;
            const dLon = (lon2 - lon1) * Math.PI / 180;
            const a = 
                Math.sin(dLat/2) * Math.sin(dLat/2) +
                Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2); 
            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); 
            return R * c; 
        }

        // Display the clinics
        function renderClinics(userLat = null, userLng = null) {
            const container = document.getElementById("clinicsContainer");
            const countLabel = document.getElementById("resultsCount");
            container.innerHTML = ""; 

            let listToRender = [...clinicsData];

            if (userLat && userLng) {
                listToRender.forEach(clinic => {
                    clinic.distance = calculateDistance(userLat, userLng, clinic.lat, clinic.lng);
                });
                // Sort by closest distance
                listToRender.sort((a, b) => a.distance - b.distance);
                countLabel.innerHTML = "🟢 GPS Active: Showing nearest clinics to you.";
            } else {

        // GPS Button Logic
        function getLocation() {
            const countLabel = document.getElementById("resultsCount");
            if (navigator.geolocation) {
                countLabel.innerText = "Requesting GPS permission...";
                navigator.geolocation.getCurrentPosition(
                    (position) => {
                        const lat = position.coords.latitude;
                        const lng = position.coords.longitude;
                        renderClinics(lat, lng);
                    },
                    (error) => {
                        countLabel.innerText = "❌ GPS Error: Location access denied. Please check your browser settings.";
                    }
                );
            } else {
                countLabel.innerText = "❌ Browser doesn't support automatic location.";
            }
        }
